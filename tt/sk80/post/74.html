}
 
class Dog extends Animal {
    function makeSound()
    {
        print "wuff";
    }
}
 
function printTheRightSound($obj)
{
    if ($obj instanceof Animal) {
        $obj->makeSound();
    } else {
        print "Error: Passed wrong kind of object";
    }
    print "\n";
}
 
printTheRightSound(new Cat());  //miau
