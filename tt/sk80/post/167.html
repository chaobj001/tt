Mysql定时备份shell

  • 2010年 03月28日 23:51
  • Mysql
#!/bin/sh
dbname=yemaosheng_com #数据库名称
backupDir=/tmp/mysqlbackup/ #备份存放路径
cd $backupDir
if [ -f $dbname".tar.gz" ]
then
        rm -rf $dbname
        rm -rf $dbname".tar.gz"
fi
mysqlhotcopy $dbname $backupDir
tar zcvf $dbname".tar.gz" "./"$dbname
<p>

保存到 /etc/cron.daily/ 目录下,加上777就行
