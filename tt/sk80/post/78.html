PHP加锁(文件与流)

  • 2010年 03月 9日 00:20
  • PHP
while(true) {
    $fp = fopen('testfile', 'w');
    echo 'Waiting for lock...';
    flock($fp, LOCK_EX);    //排他锁
    echo "OK\n";
 
    $date = date("Y-m-d H:i:s\n");
    echo $date;
    fputs($fp, $date);
    sleep(1);
 
    echo "Releasing lock...";
    flock($fp, LOCK_UN);
    echo "OK\n";
    fclose($fp);
    unsleep(1);
}
